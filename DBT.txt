import * as React from 'react';
import { LayoutSimpleHeader } from '../../../components/biz';

import { Button, Row, Col } from 'eds-react/lib/components';

import { PolicyData } from '../typing';

interface ResultPageProps extends FMS.BasicProps<any> {
    prefixCls: string;
    policyData: any;
    resultText: any;
    clickResult: () => void;
}

const dataOption = (obj, k) => {
    return (
        <Col key={k}>
            <div className="data-option">
                <div className="data-option-name">{obj.name}</div>
                <ul className="data-option-desc">
                    <li>{obj.desc[0]}</li>
                    <li>{obj.desc[1]}</li>
                    <li>{obj.desc[2]}</li>
                </ul>
                <div className="data-option-val">{obj.val}</div>
            </div>
        </Col>
    );
};

class ResultPage extends React.PureComponent<ResultPageProps> {
    constructor(props) {
        super(props);
    }
    render() {
        const { prefixCls, style, resultText } = this.props;
        return (
            <div className={`${prefixCls}__container`} style={style}>
                <div className={`${prefixCls}__cont`}>
                    <div className="info">
                        <div className="title">Available data transfer options</div>
                        <Row justify="space-around" align="middle">
                            {
                                resultText ? resultText.map((obj, k) => dataOption(obj, k)) : null
                            }
                        </Row>

                        result-page data {this.props.policyData.startTime}
                        <button onClick={this.result}>result</button>

                    </div>
                </div>
            </div>
        );
    }

    private result = () => {
        const resultData = 1;
        const policyData = Object.assign({}, this.props.policyData, { result: resultData });
        this.props.clickResult(policyData);
    }
}

export default ResultPage;
